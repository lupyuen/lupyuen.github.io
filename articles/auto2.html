  --only-matching \
  "23[0-9a-f]{6}" \
  /tmp/test.log \
  | grep -v "23007000" \
  | uniq \
  >/tmp/test.addr

#  For every address, show the corresponding line in the disassembly

  #  Skip addresses that don't match
  match=$(grep "$addr:" /tmp/nuttx.S)
  if [ "$match" != "" ]; then
    echo "----- Code Address $addr"
    grep \
      --context=5 \
      --color=auto \
      "$addr:" \
      /tmp/nuttx.S
    echo
